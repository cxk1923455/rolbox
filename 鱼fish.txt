local function findPath(root, path)
    local current = root
    for segment in string.gmatch(path, "[^/]+") do
        if not current then
            return nil
        end
        current = current:FindFirstChild(segment)
    end
    return current
end

    end
    return true
end

local RemoteCache = {}

local function resolveRemote(key)
    local definition = RemotePaths[key]
    if not definition then
        return nil
    end

    local cached = RemoteCache[key]
    if cached and cached.Parent and cached.ClassName == definition.class then
        return cached
    end

    local remote = ReplicatedStorage
    for _, part in ipairs(definition.parts) do
        remote = remote and remote:FindFirstChild(part)
    end
    if remote and remote.ClassName == definition.class then
        RemoteCache[key] = remote
        return remote
    end
    return nil
end

local Queue = {
    Items = {},
    Keys = {},
    Active = false,
    LastByKey = {},
    LastResult = "待命",
    Completed = 0,
    Failed = 0,
    Maximum = 32,
}

local Governor = {
    History = {},
    LastAt = 0,
    LastReason = "ready",
}

local function randomBetween(minimum, maximum)
    if not Config.HumanizedTiming or maximum <= minimum then
        return minimum
    end
    return minimum + math.random() * (maximum - minimum)
end

local function permitAction(now)
    if not Config.SafeMode then
        Governor.LastAt = now
        return true
    end

    local writeIndex = 1
    for readIndex = 1, #Governor.History do
        local timestamp = Governor.History[readIndex]
        if now - timestamp < 60 then
            Governor.History[writeIndex] = timestamp
            writeIndex += 1
        end
    end
    for index = #Governor.History, writeIndex, -1 do
        Governor.History[index] = nil
    end

    if now - Governor.LastAt < Config.GlobalSpacing then
        Governor.LastReason = "global spacing"
        return false
    end
    if #Governor.History >= Config.MaximumActionsPerMinute then
        Governor.LastReason = "minute budget"
        return false
    end
    Governor.LastAt = now
    Governor.History[#Governor.History + 1] = now
    Governor.LastReason = "allowed"
    return true
end

local function enqueue(key, remoteKey, mode, arguments, cooldown, delay)
    if Queue.Keys[key] or #Queue.Items >= Queue.Maximum then
        return false
    end
    Queue.Keys[key] = true
    Queue.Items[#Queue.Items + 1] = {
        Key = key,
        RemoteKey = remoteKey,
        Mode = mode,
        Arguments = arguments or {},
        Cooldown = cooldown or 0,
        DueAt = os.clock() + (delay or 0),
        Attempts = 0,
    }
    return true
end

local function removeAction(index)
    local item = table.remove(Queue.Items, index)
    if item then
        Queue.Keys[item.Key] = nil
    end
    return item
end

local function removeQueuedMatching(predicate)
    for index = #Queue.Items, 1, -1 do
        local item = Queue.Items[index]
        if predicate(item) then
            Queue.Keys[item.Key] = nil
            table.remove(Queue.Items, index)
        end
    end
end

local function processQueue(now)
    if Queue.Active then
        return
    end

    local item = Queue.Items[1]
    if not item or now < item.DueAt then
        return
    end

    local last = Queue.LastByKey[item.Key] or 0
    if now - last < item.Cooldown then
        item.DueAt = last + item.Cooldown
        return
    end

    local remote = resolveRemote(item.RemoteKey)
    if not remote then
        Queue.LastResult = "远程不可用"
        removeAction(1)
        Queue.Failed += 1
        return
    end

    if not permitAction(now) then
        item.DueAt = now + math.max(0.08, Config.GlobalSpacing)
        return
    end

    Queue.Active = true
    removeAction(1)
    task.spawn(function()
        local succeeded, result = pcall(function()
            if item.Mode == "InvokeServer" then
                return remote:InvokeServer(table.unpack(item.Arguments))
            end
            remote:FireServer(table.unpack(item.Arguments))
            return true
        end)
        Queue.Active = false
        Queue.LastByKey[item.Key] = os.clock()

        if succeeded and result ~= false then
            Queue.Completed += 1
            Queue.LastResult = "操作完成"
        else
            Queue.Failed += 1
            Queue.LastResult = "操作失败"
        end
    end)
end

local function listRods()
    local found = {}
    local known = {}
    local function add(name)
        if type(name) == "string" and name ~= "" and not known[name] then
            known[name] = true
            found[#found + 1] = name
        end
    end

    local backpack = player:FindFirstChildOfClass("Backpack")
    local character = getCharacter()
    for _, container in ipairs({ backpack, character }) do
        if container then
            for _, item in ipairs(container:GetChildren()) do
                if item:IsA("Tool") and string.find(string.lower(item.Name), "rod", 1, true) then
                    add(item.Name)
                end
            end
        end
    end

    add(statValue("rod", nil))
    add(player:GetAttribute("CurrentRod"))
    table.sort(found)
    return found
end

local function listEquipment(kind, prefix)
    local found = {}
    local known = {}
    local function add(name)
        if type(name) ~= "string" or name == "" or string.lower(name) == "none" then
            return
        end
        if prefix and string.sub(name, 1, #prefix) == prefix then
            name = string.sub(name, #prefix + 1)
        end
        if name ~= "" and not known[name] then
            known[name] = true
            found[#found + 1] = name
        end
    end

    add(statValue(kind, nil))
    local stats = getStats()
    local folder = stats and stats:FindFirstChild(kind)
    if folder then
        for _, item in ipairs(folder:GetChildren()) do
            local owned = true
            if item:IsA("NumberValue") or item:IsA("IntValue") then
                owned = item.Value > 0
            elseif item:IsA("StringValue") then
                owned = item.Value ~= "" and item.Value ~= "0"
            end
            if owned then
                add(item.Name)
            end
        end
    end

    local gui = getPlayerGui()
    local list = findPath(gui, "hud/safezone/equipment/" .. kind .. "/scroll/safezone")
    if list then
        for _, item in ipairs(list:GetChildren()) do
            if item:IsA("GuiObject") and string.lower(item.Name) ~= "template" then
                add(item.Name)
            end
        end
    end
    table.sort(found)
    return found
end

local function equippedRod()
    local character = getCharacter()
    if not character then
        return nil
    end
    for _, item in ipairs(character:GetChildren()) do
        if item:IsA("Tool") then
            local values = item:FindFirstChild("values")
            if values and values:FindFirstChild("state") then
                return item
            end
            if string.find(string.lower(item.Name), "rod", 1, true) then
                return item
            end
        end
    end
    return nil
end

local function rodValue(rod, key)
    local values = rod and rod:FindFirstChild("values")
    local value = values and values:FindFirstChild(key)
    if value and value:IsA("ValueBase") then
        return value.Value
    end
    return nil
end

        local gui = getPlayerGui()
        reel = gui and (gui:FindFirstChild("reel") or gui:FindFirstChild("Reel"))
        ObservedGUI.Reel = reel
    end
    if reel and visible(reel) then
        local bar = reel:FindFirstChild("bar") or reel:FindFirstChild("Bar")
        local fish = bar and (bar:FindFirstChild("fish") or bar:FindFirstChild("Fish"))
        local playerBar = bar and (bar:FindFirstChild("playerbar") or bar:FindFirstChild("PlayerBar"))
        if fish and playerBar then
            return reel
        end

    return nil

        Fishing.ReelFault = nil
        return true
    end
    local gui = getPlayerGui()
    if not gui then
        return true
    end
    for _, child in ipairs(gui:GetChildren()) do
        if (child.Name == "reel" or child.Name == "Reel")
            and child:IsA("ScreenGui")
            and not child.Enabled then
            local bornAt = ObservedGUI.ReelBornAt[child] or (now - 2)
            if now - bornAt > 1.5 then
                Fishing.ReelFault = "原生收线界面未启动"
                return false
            end
        end
    end
    Fishing.ReelFault = nil

local function rodStateName(value)
    local names = {
        [0] = "Destroyed",
        [1] = "Unequipped",
        [2] = "Equipping",
        [3] = "Equipped",
        [4] = "Casting",
        [5] = "Searching",
        [6] = "Luring",
        [7] = "Pre-reel",
        [8] = "Reeling",
        [9] = "Catch finished",
    }
    return names[value] or ("State " .. tostring(value))
end

local function bindRodTelemetry()
    local rod = equippedRod()
    if rod == Telemetry.Rod then
        return
    end
    if Telemetry.StateConnection then
        pcall(function()
            Telemetry.StateConnection:Disconnect()
        end)
        Telemetry.StateConnection = nil
    end
    Telemetry.Rod = rod
    Telemetry.StateSince = os.clock()
    local state = rod and rod:FindFirstChild("values") and rod.values:FindFirstChild("state")
    if state and state:IsA("ValueBase") then
        local function readState()
            Telemetry.State = rodStateName(state.Value)
            Telemetry.StateSince = os.clock()
            Telemetry.StateTransitions += 1
        end
        readState()
        Telemetry.StateConnection = state:GetPropertyChangedSignal("Value"):Connect(readState)
        Runtime.Connections[#Runtime.Connections + 1] = Telemetry.StateConnection
    else
        Telemetry.State = "No rod"
    end
end

local function setReelHeld(value)
    if Fishing.ReelHeld == value then
        return
    end
    local succeeded = pcall(function()
        VirtualInputManager:SendKeyEvent(value, Enum.KeyCode.Space, false, game)
    end)
    if succeeded then
        Fishing.ReelHeld = value
    end
end

local function resetFishing(delay)
    setReelHeld(false)
    Fishing.Phase = "Idle"
    Fishing.NextCastAt = os.clock() + (delay or 0)
    Fishing.NextReelAt = 0
    Fishing.LastFishX = nil
    Fishing.LastFishAt = 0
end

local function castArguments()
    if Config.CastProfile == "Perfect" then
        return 100, true
    end
    if Config.CastProfile == "Precision" then
        return 95, false
    end
    if Config.CastProfile == "Economy" then
        return math.random(70, 84), false
    end
    return math.random(78, 94), false
end

local function triggerShake(button, now)
    if now < Fishing.NextShakeAt then
        return
    end

    local baseDelay = math.max(Config.InputProfile == "Rapid" and 0.16 or 0.24, Config.ShakeDelay)
    local nextDelay = Config.InputProfile == "Humanized"
        and randomBetween(baseDelay, baseDelay * 1.35)
        or baseDelay
    Fishing.NextShakeAt = now + nextDelay
    task.spawn(function()
        if not button.Parent or button.AbsoluteSize.X <= 0 or button.AbsoluteSize.Y <= 0 then
            return
        end
        local point = button.AbsolutePosition + button.AbsoluteSize / 2
        pcall(function()
            VirtualInputManager:SendMouseButtonEvent(point.X, point.Y, 0, true, game, 0)
            task.wait(0.03)
            VirtualInputManager:SendMouseButtonEvent(point.X, point.Y, 0, false, game, 0)
        end)
    end)
end

local function trackReel(reel, now)
    if now < Fishing.ReelStartedAt + Config.ReelDelay or now < Fishing.NextReelAt then
        return
    end

    local bar = reel:FindFirstChild("bar") or reel:FindFirstChild("Bar")
    local fish = bar and (bar:FindFirstChild("fish") or bar:FindFirstChild("Fish"))
    local playerBar = bar and (bar:FindFirstChild("playerbar") or bar:FindFirstChild("PlayerBar"))
    if not fish or not playerBar then
        setReelHeld(false)
        Fishing.NextReelAt = now + 0.1
        return
    end

    local fishX = fish.Position.X.Scale
    local playerX = playerBar.Position.X.Scale
    local playerSize = playerBar.Size.X.Scale
    local predicted = fishX
    if Fishing.LastFishX and now > Fishing.LastFishAt then
        local elapsed = math.max(now - Fishing.LastFishAt, 0.001)
        local velocity = math.clamp((fishX - Fishing.LastFishX) / elapsed, -2.2, 2.2)
        predicted = math.clamp(fishX + velocity * 0.04, 0, 1)
    end
    Fishing.LastFishX = fishX
    Fishing.LastFishAt = now

    local deadzone = math.clamp(playerSize * 0.08, 0.012, 0.045)
    if predicted > playerX + deadzone then
        setReelHeld(true)
    elseif predicted < playerX - deadzone then
        setReelHeld(false)
    end
    Fishing.NextReelAt = now + (Config.InputProfile == "Rapid" and 0.032 or 0.045)
end

local function stepFishing(now)
    local rod = equippedRod()
    local rodState = rodValue(rod, "state")
    local reel = reelGui()
    local shake = shakeButton()

    if not reelSystemHealthy(now) then
        setReelHeld(false)
        Fishing.Phase = "收线界面异常"
        return
    end

    if reel then
        if Runtime.SetCollapsed and not Runtime.Collapsed then
            Runtime.SetCollapsed(true)
        end
        if Fishing.Phase ~= "Reeling" then
            Fishing.Phase = "Reeling"
            Fishing.ReelStartedAt = now
            Fishing.LastFishX = nil
            Fishing.LastFishAt = 0
        end
        if Config.AutoReel then
            trackReel(reel, now)
        else
            setReelHeld(false)
        end
        return
    end

    if Fishing.Phase == "Reeling" then
        resetFishing(math.random(12, 24) / 10)
        return
    end

    if shake then
        if Runtime.SetCollapsed and not Runtime.Collapsed then
            Runtime.SetCollapsed(true)
        end
        Fishing.Phase = "Shaking"
        if Config.AutoShake then
            triggerShake(shake, now)
        end
        return
    end

    if rodState and rodState >= RodState.Casting then
        Fishing.Phase = "Fishing"
        if Fishing.CastStartedAt > 0 and now - Fishing.CastStartedAt > 45 then
            Fishing.RecoveryCount += 1
            resetFishing(2)
        end
        return
    end

    if not Config.AutoCast or now < Fishing.NextCastAt then
        return
    end

    if not rod or rodState ~= RodState.Equipped or rodValue(rod, "casted") == true then
        Fishing.NextCastAt = now + 1
        return
    end

    local power, perfect = castArguments()
    if enqueue("cast", "Cast", "InvokeServer", { power, perfect }, Config.CastCooldown) then
        Fishing.Phase = "Casting"
        Fishing.CastStartedAt = now
        Fishing.NextCastAt = now + Config.CastCooldown
    end
end

local function equipBobber(name)
    if type(name) ~= "string" or name == "" then
        return false

end

local function stepAutoEquip(now)
    if not Config.AutoEquip or now < Equipment.NextCheckAt then
        return
    end
    if now < Equipment.ManualOverrideUntil then
        return
    end
    Equipment.NextCheckAt = now + 2
    local targetRod = Config.TargetRod
    local rod = equippedRod()
    if targetRod and (not rod or rod.Name ~= targetRod) then
        equipRod(targetRod)
    end
    if Config.TargetBait and statValue("bait", nil) ~= Config.TargetBait then
        equipBait(Config.TargetBait)
    end
    if Config.TargetBobber and statValue("bobber", nil) ~= Config.TargetBobber then
        equipBobber(Config.TargetBobber)
    end
end

local function syncSellRules(forceDisabled)
    removeQueuedMatching(function(item)
        return string.sub(item.Key, 1, 9) == "sellrule_"
    end)
    for name, setting in pairs(SellRules) do

local Freeze = {
    Snapshot = nil,
    NextAt = 0,
}

local function restoreFreeze()
    local snapshot = Freeze.Snapshot
    Freeze.Snapshot = nil
    if not snapshot then
        return
    end
    pcall(function()
        if snapshot.Humanoid and snapshot.Humanoid.Parent then
            snapshot.Humanoid.WalkSpeed = snapshot.WalkSpeed
            snapshot.Humanoid.JumpPower = snapshot.JumpPower
            snapshot.Humanoid.UseJumpPower = snapshot.UseJumpPower
            snapshot.Humanoid.JumpHeight = snapshot.JumpHeight
            snapshot.Humanoid.AutoRotate = snapshot.AutoRotate
        end
        if snapshot.Root and snapshot.Root.Parent then
            snapshot.Root.Anchored = snapshot.Anchored
        end
    end)
end

local function stepFreeze(now)
    if not Config.FreezeCharacter then
        restoreFreeze()
        return
    end
    if now < Freeze.NextAt then
        return
    end
    Freeze.NextAt = now + 0.2
    local character = getCharacter()
    local humanoid = character and character:FindFirstChildOfClass("Humanoid")
    local root = character and character:FindFirstChild("HumanoidRootPart")
    if not humanoid or not root then
        return
    end
    if not Freeze.Snapshot or Freeze.Snapshot.Humanoid ~= humanoid then
        restoreFreeze()
        Freeze.Snapshot = {
            Humanoid = humanoid,
            Root = root,
            WalkSpeed = humanoid.WalkSpeed,
            JumpPower = humanoid.JumpPower,
            JumpHeight = humanoid.JumpHeight,
            UseJumpPower = humanoid.UseJumpPower,
            AutoRotate = humanoid.AutoRotate,
            Anchored = root.Anchored,
        }
    end
    humanoid.WalkSpeed = 0
    humanoid.JumpPower = 0
    humanoid.AutoRotate = false
    root.Anchored = true
end

local Oxygen = {
    Snapshot = nil,
    NextAt = 0,
}

local function restoreOxygen()
    local snapshot = Oxygen.Snapshot

        end

local function stepOxygen(now)
    if not Config.InfiniteOxygen then
        restoreOxygen()
        return
    end
    if now < Oxygen.NextAt then
        return
    end
    Oxygen.NextAt = now + 0.5
    local character = getCharacter()
    local tank = character and character:FindFirstChild("DivingTank")

            tank:SetAttribute("Tier", 9999)
        end
    end)

local Discovery = {
    Pending = {},
    NextAt = 0,
    Total = 0,
    Completed = 0,
}

local function queueLocationDiscovery()
    local map = findPath(workspace, "world/map")
    if not map or not resolveRemote("Discover") then
        return false
    end
    local names = {}
    local known = {}
    local function add(name)
        if type(name) == "string" and name ~= "" and not known[name] then
            known[name] = true
            names[#names + 1] = name
        end
    end
    for _, location in ipairs(map:GetChildren()) do
        if location:IsA("Folder") then
            for _, child in ipairs(location:GetChildren()) do
                add(child.Name)
            end
        else
            add(location.Name)
        end
    end
    table.sort(names)
    Discovery.Pending = names
    Discovery.Total = #names
    Discovery.Completed = 0
    Discovery.NextAt = os.clock()
    return #names > 0
end

local function stepDiscovery(now)
    if #Discovery.Pending == 0 or now < Discovery.NextAt then
        return
    end
    if #Queue.Items >= math.floor(Queue.Maximum / 2) then
        return
    end
    local name = table.remove(Discovery.Pending, 1)
    if enqueue("discover_" .. name, "Discover", "FireServer", { name }, 0.45) then
        Discovery.Completed += 1
        Discovery.NextAt = now + randomBetween(0.55, 0.9)
    else
        table.insert(Discovery.Pending, 1, name)
    end
end

local function spawnPosition(name)
    local spawns = findPath(workspace, "world/spawns")
    local spawn = spawns and spawns:FindFirstChild(name or "")
    if spawn and spawn:IsA("BasePart") then
        return spawn.Position
    end
    if spawn and spawn:IsA("Model") then
        return spawn:GetPivot().Position
    end
    return nil
end

local function clearWaypoint()
    if Navigation.Marker then
        pcall(function()
            Navigation.Marker:Destroy()
        end)
    end
    Navigation.Marker = nil
end

local function setWaypoint(name)
    local position = spawnPosition(name)
    if not position then
        return false
    end
    clearWaypoint()
    local marker = Instance.new("Part")
    marker.Name = "FischControlWaypoint"
    marker.Anchored = true
    marker.CanCollide = false
    marker.CanQuery = false
    marker.CanTouch = false
    marker.Material = Enum.Material.Neon
    marker.Color = Color3.fromRGB(78, 181, 255)
    marker.Transparency = 0.25
    marker.Size = Vector3.new(2, 12, 2)
    marker.CFrame = CFrame.new(position + Vector3.new(0, 6, 0))
    marker.Parent = workspace
    Navigation.Target = name
    Navigation.Marker = marker
    return true
end

local function distanceToWaypoint()
    local position = spawnPosition(Navigation.Target)
    local character = getCharacter()
    local root = character and character:FindFirstChild("HumanoidRootPart")
    if not position or not root then
        return nil
    end
    return math.floor((root.Position - position).Magnitude)
end

            BorderSizePixel = 0,
            Position = UDim2.new(1, x, 0.5, 0),

    end

        AnchorPoint = Vector2.new(1, 0.5),

        setter(math.max(minimum, getter() - delta))
        refresh()
    end))
    track(increase.Activated:Connect(function()

local function updateUI(now)
    if now < (Runtime.NextUIAt or 0) then
        return
    end
    Runtime.NextUIAt = now + 0.35
    if not UI.Root or not UI.Root.Parent then
        return
    end
    if Runtime.ApplyResponsiveLayout then
        Runtime.ApplyResponsiveLayout()
    end

    for _, refresh in ipairs(UI.Dynamic) do
        pcall(refresh)
    end

    local state = UI.Labels.State
    local stats = UI.Labels.Stats
    local queue = UI.Labels.Queue
    local health = UI.Labels.Error
    if state then
        if Runtime.Ready then
            state.Text = "状态：" .. displayPhase(Fishing.Phase) .. " | 恢复次数：" .. Fishing.RecoveryCount
        else
            state.Text = "状态：等待游戏数据和资源"
        end
    end
    if stats then
        stats.Text = ("数据：等级 %s | 金币 %s | %s"):format(
            tostring(statValue("level", "?")),
            tostring(statValue("coins", "?")),
            tostring(statValue("rod", "未装备鱼竿"))
        )
    end
    if queue then
        queue.Text = ("队列：%d 等待 | %d 成功 | %d 失败"):format(
            #Queue.Items,
            Queue.Completed,
            Queue.Failed
        )
    end
    if health then
        if Fishing.ReelFault then
            health.Text = "收线界面异常：请重新进入游戏后再钓鱼"
            health.TextColor3 = Color3.fromRGB(245, 126, 104)
        elseif Runtime.ErrorCount > 0 then
            health.Text = "健康：已恢复错误 " .. Runtime.ErrorCount
            health.TextColor3 = Color3.fromRGB(245, 181, 90)
        else
            health.Text = "健康：正常 | " .. Queue.LastResult
            health.TextColor3 = Color3.fromRGB(106, 211, 163)
        end
    end
    if UI.Labels.Loadout then
        UI.Labels.Loadout.Text = ("装备：竿 %s | 饵 %s | 标 %s"):format(
            tostring(statValue("rod", "none")),
            tostring(statValue("bait", "none")),
            tostring(statValue("bobber", "none"))
        )
    end
    if UI.Labels.Capabilities then
        local cast = resolveRemote("Cast") and "抛竿" or "抛竿缺失"
        local gear = resolveRemote("EquipRod") and "装备" or "装备缺失"
        local settings = resolveRemote("Settings") and "规则" or "规则缺失"
        UI.Labels.Capabilities.Text = "协议能力：" .. cast .. " / " .. gear .. " / " .. settings
    end
    if UI.Labels.Discovery then
        if #Discovery.Pending > 0 then
            UI.Labels.Discovery.Text = ("地点任务：%d/%d 已排队"):format(
                Discovery.Completed,
                Discovery.Total
            )
        else
            UI.Labels.Discovery.Text = "地点任务：空闲"
        end
    end
    if UI.Labels.Character then
        local character = getCharacter()
        local humanoid = character and character:FindFirstChildOfClass("Humanoid")
        local healthValue = humanoid and math.floor(humanoid.Health) or 0
        UI.Labels.Character.Text = ("角色：%d HP | 氧气%s | 冻结%s"):format(
            healthValue,
            Config.InfiniteOxygen and "保持" or "正常",
            Config.FreezeCharacter and "开" or "关"
        )
    end
    if UI.Labels.Telemetry then
        local elapsed = math.floor(now - Telemetry.StateSince)
        local caught = (tonumber(statValue("tracker_fishcaught", 0)) or 0) - Telemetry.SessionFishStart
        UI.Labels.Telemetry.Text = ("鱼竿遥测：%s %d 秒 | 本局鱼 %d"):format(
            displayPhase(Telemetry.State),
            elapsed,
            caught
        )
    end
    if UI.Labels.WorldInfo then
        local character = getCharacter()
        local zone = character and character:GetAttribute("zone")
        local zoneValue = character and character:FindFirstChild("zone")
        if zoneValue and zoneValue:IsA("ValueBase") then
            zone = zoneValue.Value
        end
        local weatherFolder = findPath(workspace, "world/weather")
        local weather = "晴朗"
        if weatherFolder then
            local active = weatherFolder:GetChildren()[1]
            weather = active and active.Name or weather
        end
        local day = workspace:GetAttribute("CurrentDayNumber") or "?"
        local activeEvent = nil
        for _, attribute in ipairs({
            "WhirlpoolActive", "MarianasVeilActive", "ScyllaHuntIsActive", "ReefTitan_IsActive",
        }) do
            if workspace:GetAttribute(attribute) == true then
                activeEvent = attribute:gsub("Active", "")
                break
            end
        end
        local waypoint = Navigation.Target and (" | 航点 " .. Navigation.Target .. " " .. tostring(distanceToWaypoint() or "?") .. " 格") or ""
        UI.Labels.WorldInfo.Text = ("世界：%s | %s | 第 %s 天%s%s"):format(
            tostring(zone or "未知区域"),
            weather,
            tostring(day),
            activeEvent and (" | " .. activeEvent) or "",
            waypoint
        )
    end
end

function Runtime.Unload()
    if Runtime.Unloaded then
        return
    end
    Runtime.Unloaded = true
    Runtime.Alive = false
    setReelHeld(false)
    restoreFreeze()
    restoreOxygen()
    clearWaypoint()
    Discovery.Pending = {}
    Queue.Items = {}
    Queue.Keys = {}
    disconnectAll()
    destroyUI()
    if env.FISCH_CONTROL_STUDIO == Runtime then
        env.FISCH_CONTROL_STUDIO = nil
    end
end

track(RunService.Heartbeat:Connect(function()
    if not Runtime.Alive then
        return
    end
    local now = os.clock()
    if now - Runtime.LastTickAt < 0.08 then
        return
    end
    Runtime.LastTickAt = now

    local succeeded, errorMessage = xpcall(function()
        Runtime.Ready = clientReady()
        if not Runtime.Ready then
            updateUI(now)
            return
        end
        processQueue(now)
        initializeLoadout()
        bindRodTelemetry()
        stepFishing(now)
        stepAutoEquip(now)
        stepAutoSell(now)
        stepFreeze(now)
        stepOxygen(now)
        stepDiscovery(now)
        if (reelGui() or shakeButton()) and Runtime.SetCollapsed and not Runtime.Collapsed then
            Runtime.SetCollapsed(true)
        end
        updateUI(now)
    end, debug.traceback)
    if not succeeded then
        Runtime.ErrorCount += 1
        Runtime.LastError = tostring(errorMessage)
    end
end))
